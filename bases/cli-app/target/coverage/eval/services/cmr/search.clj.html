001  (ns eval.services.cmr.search
002    (:require
003     [eval.cmr.commands.search :as search-api]
004     [eval.cmr.core :as cmr]))
005  
006  (defn search
007    "Search concepts for concepts"
008    [client concept-type query & [opts]]
009    (cmr/decode-cmr-response-body
010     (cmr/invoke client (search-api/search concept-type query opts))))
011  
012  (defn query-hits
013    "Query CMR for count of concepts available that are available from
014    a given query.
015  
016    Takes a query and sets a :page_size of 0 and returns
017    the CMR-Hits header string as an integer value."
018    [client concept-type query & [opts]]
019    (let [query (-> query
020                    (as-> q (search-api/search concept-type q opts))
021                    (assoc :page_size 0))]
022      (-> (cmr/invoke client query)
023          (get-in [:headers :CMR-Hits])
024          Integer/parseInt)))
