001  (ns eval.services.cmr.bulk.granule
002    (:require
003     [eval.cmr.commands.bulk-granule :as bulk-granule]
004     [eval.cmr.core :as cmr]
005     [taoensso.timbre :as log])
006    (:import
007     (java.time Instant)))
008  
009  (defn submit-job!
010    "POST a bulk granule update job to CMR and return the response."
011    [client provider job-def]
012    (let [job (cmr/decode-cmr-response-body
013               (cmr/invoke client (bulk-granule/post-job provider job-def)))]
014      (log/info (format "Bulk-Granule-Update job created with ID [%s]" (:task-id job)))
015      job))
016  
017  (defn trigger-status-update!
018    "Trigger an update of bulk granule job statuses."
019    [client & [opts]]
020    (cmr/decode-cmr-response-body
021     (cmr/invoke client (bulk-granule/trigger-update opts))))
022  
023  (defn fetch-job-status
024    "Request bulk granule update job status from CMR."
025    [client job-id & [opts]]
026  
027    (let [{:keys [show-granules
028                  show-progress
029                  show-request]} opts
030          query-params {:show_granules (or show-granules false)
031                        :show_progress (or show-progress false)
032                        :show_request (or show-request false)}]
033      (cmr/decode-cmr-response-body
034       (cmr/invoke client (bulk-granule/get-job-status job-id query-params)))))
035  
036  (defn benchmark
037    "Request status with a delay to compute per-second updates happening
038    in the bulk granule update job.
039    TODO: make an async, non-blocking version"
040    ([client task-id]
041     (benchmark client task-id 1))
042    ([client task-id time-in-sec]
043     (let [resp (cmr/decode-cmr-response-body
044                 (cmr/invoke client
045                             (bulk-granule/get-job-status
046                              task-id
047                              {:show_granules true})))
048           get-counts #(->> resp
049                            :granule-statuses
050                            (map :status)
051                            frequencies)
052           start-counts (get-counts)
053           ;; Note where start and end are calculated from
054           ;; The query takes a while so we clock it from when it was
055           ;; sent, not when the response comes back
056           start-time (Instant/now)
057  
058           _ (Thread/sleep (* 1000 time-in-sec))
059  
060           end-time (Instant/now)
061           end-counts (get-counts)
062           duration (quot (- (.toEpochMilli end-time)
063                             (.toEpochMilli start-time))
064                          1000)]
065       {:task-id task-id
066        :start-time start-time
067        :end-time end-time
068        :benchmark-duration duration
069        :start-counts start-counts
070        :end-counts end-counts
071        :processed (- (get start-counts "PENDING" 0)
072                      (get end-counts "PENDING" 0))})))
073  
074  (defn add-update-instructions
075    "Add a list of update instructions to a bulk granule update request.
076    based on a transformation."
077    [job urs xf]
078    (->> urs
079         (map (fn [id] [id (xf id)]))
080         (assoc job :updates)))
