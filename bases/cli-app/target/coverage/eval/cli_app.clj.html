001  (ns eval.cli-app
002    (:gen-class)
003    (:require
004     [clojure.spec.alpha :as spec]
005     [clojure.string :as str]
006     [eval.services.cmr.search :as search]))
007  
008  (spec/def ::arg-list (spec/cat :command string?
009                                 :cmr keyword?
010                                 :query-params (spec/? map?)
011                                 :options (spec/? map?)))
012  
013  (defn command-handler
014    [context args]
015    (println args)
016    (let [args (str/split args #" ")
017          command (first args)
018          client-id (keyword (second args))
019          client (get-in context [:instances client-id])
020          args (drop 2 args)]
021      (when-not client
022       {::fault (ex-info "Invalid command"
023                         {:error "No such client [" client-id "]"
024                          :context context})})
025      (case command
026        "search" (search/search client (keyword (first args)) {} {:format :umm-json})
027        (do
028          (println "unrecognized command [" command "]")
029          {:exit true}))))
030  
031  (defn create-app
032    [cfg]
033    (loop []
034      (flush)
035      (print "Enter: ")
036      (let [in (str/trim (read-line))]
037        (case in
038          "quit" :quit
039          "help" (do (println "help coming soon, \"quit\" to quit")
040                     (recur))
041          (do
042            (command-handler cfg in)
043            (recur))))))
