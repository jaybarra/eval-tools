001  (ns eval.api.cmr
002    (:require
003     [eval.services.cmr.providers :as providers]
004     [ring.util.response :refer [response not-found]]))
005  
006  (defn cmr-not-found-wrapper
007    "Middleware to check a given cmr-inst exists in the system"
008    [handler]
009    (fn [{:keys [cmr path-params] :as request}]
010      (let [cmr-inst (keyword (get path-params :cmr-inst))]
011        (if-not (get-in cmr [:instances cmr-inst])
012          (not-found {:error "No such CMR instance"
013                      :id cmr-inst})
014          (handler request)))))
015  
016  (defn get-config
017    "Return the configured CMR clients"
018    [request]
019    (let [cmr (get-in request [:cmr :instances])]
020      (response cmr)))
021  
022  (defn get-providers
023    "Return a list of providers from a CMR instance"
024    [request]
025    (let [cmr-inst (get-in request [:path-params :cmr-inst])
026          client (get-in request [:cmr :instances (keyword cmr-inst)])]
027      (response (providers/get-providers client))))
028  
029  (def routes
030    ["/cmr"
031     ["/overview" {:handler get-config}]
032     ["/:cmr-inst" {:middleware [cmr-not-found-wrapper]}
033      ["/providers" {:handler get-providers}]]])
