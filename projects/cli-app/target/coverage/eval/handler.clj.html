016  (defn cors-middleware-wrapper
017    [handler]
018    (fn [request]
019      (-> request
020          handler
021          (assoc-in [:headers "Access-Control-Allow-Origin"] "*"))))
022  
023  (def cors-middleware
024    {:name ::cors
025     :description "Apply CORS headers to responses"
026     :wrap cors-middleware-wrapper})
027  
028  (defn cmr-middlware-wrapper
029    [handler cmr]
030    (fn [request]
031      (handler (assoc request :cmr cmr))))
032  
033  (def cmr-middleware
034    {:name ::cmr
035     :description "add cmr context to requests"
036     :wrap cmr-middlware-wrapper})
037

042  (defn local-error-handler
043    [message ex request]
044    {:status 500
045     :body {:message message
046            ;; :exception (.getClass ex)
047            :data (ex-data ex)

049  
050  (def cmr-exception-middleware
051    (exception/create-exception-middleware

053      exception/default-handlers
054      {:cmr (partial local-error-handler "CMR Error")})))
