001  (ns eval.utils.core
002    (:require
003     [clojure.spec.alpha :as s]
004     [clojure.math.numeric-tower :as math]))
005  
006  (set! *warn-on-reflection* true)
007  
008  (s/def ::positive-num (comp not neg?))
009  
010  (defn spec-problems
011    [spec value]
012    (:clojure.spec.alpha/problems (s/explain-data spec value)))
013  
014  (defn spec-validate
015    "Check spec by value.
016    Returns nil on pass, otherwise throws an exception"
017    [spec value]
018    (when-let [problem (->> (spec-problems spec value)
019                            first
020                            :via)]
021      (throw (ex-info "Validation failed" {:validation problem}))))
022  
023  (defn within?
024    "Returns true if two numbers are nearly equal within a given delta.
025    Should only be used for 6 significant digits or fewer.
026    Eg (within? 0.1 2.4 2.41) => true"
027    [^Double delta a b]
028    (spec-validate ::positive-num delta)
029    (<= (- (double (math/abs (- (double a) (double b)))) 1e-7)
030        (double delta)))
