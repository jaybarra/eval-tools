001  (ns eval.db.document-store
002    (:require
003     [clojure.java.io :as io]
004     [crux.api :as crux]
005     [taoensso.timbre :as log]))
006  
007  (defprotocol DocumentStore
008    (save! [this document])
009    (query [this query])
010    (halt! [this]))
011  
012  (defrecord CruxStore [crux-node]
013  
014    DocumentStore
015  
016    (save! [this document]
017      (crux/submit-tx (:crux-node this) [[:crux.tx/put document]]))
018  
019    (query [this query]
020      (crux/q (crux/db (:crux-node this)) query))
021  
022    (halt! [this]
023      (when-let [node (:crux-node this)]
024        (.close node)
025        (dissoc this :crux-node))))
026  
027  (defmulti create-document-store :type)
028  
029  (defmethod create-document-store :default
030    [_]
031    ;; TODO convert to in-memory store
032    (log/warn "NOOP document store, no persistence")
033    (reify DocumentStore
034      (save! [this document] (log/debug "NOOP save document" document))
035      (query [this query] (log/debug "NOOP query" query))
036      (halt! [this] nil)))
037  
038  (defmethod create-document-store :crux
039    [opts]
040    (letfn [(kv-store [dir]
041              {:kv-store {:crux/module 'crux.rocksdb/->kv-store
042                          :db-dir (io/file dir)
043                          :sync? true}})]
044      (->CruxStore (crux/start-node
045                    {:crux/tx-log (kv-store (get opts :log-dir "data/dev/tx-log"))
046                     :crux/document-store (kv-store (get opts :doc-dir "data/dev/doc-store"))
047                     :crux/index-store (kv-store (get opts :index-dir "data/dev/index-store"))}))))
048  
049  (defn stop-document-store
050    [store]
051    (halt! store))
